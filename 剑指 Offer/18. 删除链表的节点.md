## 遍历

- 时间复杂度：O(n)
- 空间复杂度：O(1)

```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* deleteNode(ListNode* head, int val) {
        if (head->val == val) return head->next;

        ListNode* pre = nullptr, * cur = head;
        while (cur-> val != val) {
            pre = cur;
            cur = cur->next;
        }
        pre->next = cur->next;
        return head;
    }
};
```
